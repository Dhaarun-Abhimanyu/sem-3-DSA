#include <stdio.h>
#include <stdlib.h>

int main(){
    int n,i;
    printf("\nNo. of elements : ");
    scanf("%d",&n);
    int a[n];
    printf("\nEnter %d elements:\n",n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    int left=0,right=n-1,min=a[right];
    while(left<=right){//4 5 6 7 1 2 3
        int mid = (left+right)/2;
        if(a[mid] > a[right])
            left = mid+1;
        else if(a[mid] > a[left])
            right = mid-1;
        else{
            right = mid-1;
            if(a[mid] < min){
                min = a[mid];
            }
        }
        }
        printf("\nPivot : %d",min);

return 0;
}
